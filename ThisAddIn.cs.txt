// THISADDIN.CS
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Xml.Linq;
using Excel = Microsoft.Office.Interop.Excel;
using Office = Microsoft.Office.Core;
using Microsoft.Office.Tools.Excel;

namespace AddInVSTO
{
    public partial class ThisAddIn
    {

        private void ThisAddIn_Startup(object sender, System.EventArgs e)
        {
            try
            {
                // Например, подключение событий Excel
                this.Application.WorkbookOpen += Application_WorkbookOpen;

                // Автоматическая активация чекбокса при запуске
                var customRibbon = Globals.Ribbons.GetRibbon<UserCustomRibbon>();
                if (customRibbon != null)
                {
                    customRibbon.EnableCheckBox = true;
                }
            }
            catch (Exception ex)
            {
                System.Diagnostics.Debug.WriteLine("Ошибка при запуске: " + ex.Message);
            }
        }

        private void Application_WorkbookOpen(Excel.Workbook wb)
        {
            System.Diagnostics.Debug.WriteLine("Открыта книга: " + wb.Name);
        }

        private void ThisAddIn_Shutdown(object sender, System.EventArgs e)
        {
        }

        protected override Microsoft.Office.Core.IRibbonExtensibility CreateRibbonExtensibilityObject()
        {
            return new UserCustomRibbon();
        }
        


        #region Код, автоматически созданный VSTO

        /// <summary>
        /// Требуемый метод для поддержки конструктора — не изменяйте 
        /// содержимое этого метода с помощью редактора кода.
        /// </summary>
        private void InternalStartup()
        {
            this.Startup += new System.EventHandler(ThisAddIn_Startup);
            this.Shutdown += new System.EventHandler(ThisAddIn_Shutdown);
        }

        #endregion
    }
}
